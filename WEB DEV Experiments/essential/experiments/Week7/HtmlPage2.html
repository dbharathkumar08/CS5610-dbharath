Code Below will show JSON object manipulation

 var employees = [
            {
                "firstName": "Bharath",
                "lastName": "Kumar"
            },
            {
                "firstName": "Murali",
                "lastName": "Vijay"
            },
            {
                "firstName": "Dinesh",
                "lastName": "Mongia"
            },
        ];
        
        document.getElementById("demo").innerHTML =
        employees[0].firstName + " " + employees[0].lastName;
        document.getElementById("demo1").innerHTML =
        employees[1].firstName + " " + employees[1].lastName;
        document.getElementById("demo2").innerHTML =
        employees[2].firstName + " " + employees[2].lastName;
        employees[0].firstName = "Kishor";
        employees[1]["lastName"] = "Karthik";
        employees[1]["firstName"] = "Nayan";
        document.getElementById("demo3").innerHTML =
        employees[0].firstName + " " + employees[0].lastName;
        document.getElementById("demo4").innerHTML =
        employees[1].firstName + " " + employees[1].lastName;
        document.getElementById("demo5").innerHTML =
        employees[2].firstName + " " + employees[2].lastName;
